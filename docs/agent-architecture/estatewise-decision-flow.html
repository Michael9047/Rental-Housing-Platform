// Handoff 协议核心逻辑 class HandoffManager { maxDepth = 3; visited = new Set<string>(); // 环路检测 async handoff(from: Agent, to: Agent, task: Task) { if (depth >= maxDepth) throw 'Max handoff depth'; if (visited.has(to.id)) throw 'Cycle detected'; if (!await healthCheck(to)) throw 'Target unhealthy'; visited.add(to.id); return await to.execute(task); // 目标 Agent 进入自己的 ReAct Loop } }
